有一个双生怪
1.血量不可见
2.出生时血量一样
3.怪物最高100血
4.血量不共享
5.每一次可以攻击任意血量的血
6.打到任何一支血量剩余一点，则任务成功成功
请问有15次攻击机会怎么使用保证100%成功？

思路为记录每一次给怪物造成的伤害,同时让一只怪的总伤害减去另一只总伤始终少于剩余的攻击次数

M2-M1<AttackNumber

如果一只怪物收到了一次攻击后死亡,那就代表他的血量在另一只怪物的血量所剩的刀数之和之间

M2 < DeadM < M2+Attack

```c#
  public static void Attack()
        {
            Random r = new Random();
            int attNumber = 15;
            int random = r.Next(1, 101);
            int monster = random;
            int monster2 = random;

      		//记录怪物收到的总伤害   最重要!!!
            int monster1Damage = 0;
            int monster2Damage = 0;
            WriteLine($"怪物血量{monster}");
            int tempAttack = 0;

            while (true)
            {
             	// 第二只怪受到的伤害减去第一只怪受到的伤害,代表需要补平血量的伤害,再加上剩余的刀数,等于当次可以造成的最大伤害
                tempAttack = monster2Damage - monster1Damage + attNumber;
                monster -= tempAttack;
                monster1Damage += tempAttack;
                WriteLine($"monster1受到 {tempAttack}点伤害 总伤害为:{random - monster}");
				//当某一只死亡时,进入只对另一只每次1伤的循环,直至血量为1
                if (monster <= 0)
                {
                    WriteLine($"monster1当前{monster.ToString()}, {attNumber}刀进入");
                    Final(monster2, attNumber);
                    return;
                }
				//出刀后减少数量
                attNumber -= 1;
                if (attNumber <= 0)
                {
                    WriteLine("刀超了");
                    return;
                }
				// 第一只怪受到的伤害减去第二只怪受到的伤害,代表需要补平血量的伤害,再加上剩余的刀数,等于当次可以造成的最大伤害
                tempAttack = (monster1Damage - monster2Damage + attNumber);
                monster2 -= tempAttack;
                monster2Damage += tempAttack;
                WriteLine($"monster2受到 {tempAttack}点伤害, 总伤害为:{random - monster2}");
                if (monster2 <= 0)
                {
                    WriteLine($"monster2当前{monster2.ToString()} ,{attNumber}刀进入");
                    Final(monster, attNumber);
                    return;
                }

                attNumber -= 1;

                if (attNumber <= 0)
                {
                    WriteLine("刀超了");
                    return;
                }
            }
        }
		//其中一只死亡时,进入每次一点伤害循环,直到怪物剩余1血
        public static void Final(int mon, int attck)
        {
            while (mon > 1)
            {
                mon -= 1;
                attck -= 1;
            }

            WriteLine($"monster 血量为 {mon},还剩 {attck} 刀,任务完成");
        }
```
